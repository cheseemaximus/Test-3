<script>
  const cart = [];

  function addToCart(item, price) {
    cart.push({ item, price });
    updateCart();
  }

  function updateCart() {
    const cartItems = document.getElementById('cart-items');
    const cartTotal = document.getElementById('cart-total');
    cartItems.innerHTML = '';
    let total = 0;

    cart.forEach((entry, index) => {
      total += entry.price;
      const div = document.createElement('div');
      div.innerHTML = `<strong>${entry.item}</strong> - $${entry.price.toFixed(2)} <button onclick="removeFromCart(${index})" style="margin-left:10px;">âŒ</button>`;
      cartItems.appendChild(div);
    });

    cartTotal.textContent = total.toFixed(2);
  }

  function removeFromCart(index) {
    cart.splice(index, 1);
    updateCart();
  }

  // Setup PayPal buttons
  paypal.Buttons({
    createOrder: (data, actions) => {
      const total = cart.reduce((sum, item) => sum + item.price, 0);
      if (!total) return alert("Your cart is empty!");
      const username = document.getElementById("roblox-username").value.trim();
      if (!username) return alert("Please enter your Roblox username.");
      return actions.order.create({
        purchase_units: [{
          amount: {
            value: total.toFixed(2)
          },
          description: `Grow a Garden Order for ${username}`
        }]
      });
    },
    onApprove: (data, actions) => {
      return actions.order.capture().then(details => {
        document.getElementById("thank-you-msg").style.display = "block";
        cart.length = 0;
        updateCart();
        document.getElementById("roblox-username").value = "";
      });
    }
  }).render('#paypal-button-container');
</script>
